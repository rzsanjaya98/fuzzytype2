  public function Upper_trapmf($x, $a, $b, $c, $d){
    if($x>$a AND $x<$b){
      $x = ($x-$a)/($b-$a);
    }else if($x>=$b AND $x<=$c){
      $x = 1;
    }else if($x>$c AND $x<$d){
      $x = ($d-$x)/($d-$c);
    }else{
      $x = 0;
    }

    return $x;
  }

  public function Lower_trapmf($x, $a, $b, $c, $d){
    if($x>$a AND $x<$b){
      $x = ($x-$a)/($b-$a);
    }else if($x>=$b AND $x<=$c){
      $x = 0.8;
    }else if($x>$c AND $x<$d){
      $x = ($d-$x)/($d-$c);
    }else{
      $x = 0;
    }

    return $x;
  }
